```c++
class MinStack {
public:
    stack<int> s;
    stack<int> m;
    /** initialize your data structure here. */
    MinStack() {
        
    }
    
    void push(int x) {
        s.push(x);
        int currentMin = (!m.empty() && m.top() <  x) ? m.top() : x;
        m.push(currentMin);
    }
    
    void pop() {
        s.pop();
        m.pop();
    }
    
    int top() {
        return s.top();
    }
    
    int getMin() {
        return m.top();
    }
};
```
