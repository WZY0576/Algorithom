```c++
#include<stack>
#include<stdio.h>
using namespace std;
int Q;
char C[3];
int k;
stack<int> left;
stack<int> right;
stack<int> sum;
int m[(int)1e6];
int top = -1;

int main() {
    scanf("%d", &Q);
    while(Q--) {
        scanf("%s", C);
        if(C[0] == 'I') {
            scanf("%d", &k);
            left.push(k);
            int currentSum = sum.empty() ? k : sum.top() + k;
            sum.push(currentSum);
            if(top == -1 || currentSum > m[top]) {
                m[++top] = currentSum;
            } else {
                int temp = m[top];
                m[++top] = temp;
            }
            continue;
        }
        if(C[0] == 'D') {
            if(!left.empty()) {
                left.pop();
                sum.pop();
                top--;
            }
            continue;
        }
        if(C[0] == 'L') {
            if(!left.empty()) {
                right.push(left.top());
                left.pop();
                sum.pop();
                top--;
            }
            continue;
        }
        if(C[0] == 'R') {
            if(!right.empty()) {
                int delta = right.top();
                right.pop();
                left.push(delta);
                int currentSum = sum.empty() ? delta : sum.top() + delta;
                sum.push(currentSum);
                if(top == -1 || currentSum > m[top]) {
                    m[++top] = currentSum;
                } else {
                    int temp = m[top];
                    m[++top] = temp;
                }
            }
            continue;
        }
        if(C[0] == 'Q') {
            scanf("%d", &k);
            if(top >= 0)    printf("%d\n", m[k-1]);
            else    printf("%d\n", 0);
            continue;
        }
    }
    return 0;
}
```
